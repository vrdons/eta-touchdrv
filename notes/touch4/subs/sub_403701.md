# sub_403701

## Control flow

Builds encoded output packet from two interval lists. It zeroes output state bytes at `[0]` and `[1]`, sets output length word at `+0x28` to `2`, then walks both sorted `uint16_t start/end` range arrays until position exceeds `0x280`. It emits run bytes into output body. If pending gap exceeds `0x40`, it emits `0xfc` or `0xfd` depending on active side and subtracts `0x40`; otherwise it emits `((gap - 1) * 4) | side_bit`, advances output length, and clears pending gap. It alternates ranges from first and second list based on next start/end positions.

Called by [[sub_403a67]].

    unsigned short pos_b = 0;
    unsigned short last = 0;
    unsigned short gap = 0;
    unsigned char ia = 0;
    unsigned char ib = 0;
    uint64_t a_open = 0;
    uint64_t b_open = 0;
    uint64_t active = 0;

        uint64_t new_active = (a_open != 0 || b_open != 0) ? 1 : 0;

            while (gap != 0 && *(uint16_t *)(out + 0x28) <= 0x27) {

                    out[*(uint16_t *)(out + 0x28)] = active != 0 ? 0xfd : 0xfc;
                    gap -= 0x40;
                    (*(uint16_t *)(out + 0x28))++;
                } else {
                    out[*(uint16_t *)(out + 0x28)] = (unsigned char)(((gap - 1) * 4) + (active != 0));
                    gap = 0;
                    (*(uint16_t *)(out + 0x28))++;
                }
            }
            active = new_active;
        }

            gap += pos_a - last;
            last = pos_a;
            a_open = 1;
            if (ia < a_count) {
                pos_a = a[ia * 2 + 0];
            } else {
                pos_a = 0x281;
                a_open = 0;
            }
        } else {
            gap += pos_b - last;
            last = pos_b;
            b_open = 1;
            if (ib < b_count) {
                pos_b = b[ib * 2 + 0];
            } else {
                pos_b = 0x281;
                b_open = 0;
            }
        }

        if (pos_b >= pos_a) {
            if (a_open == 0) {
                if (ia < a_count) {
                    pos_a = a[ia * 2 + 0];
                    a_open = 1;
                } else {
                    pos_a = 0x281;
                    a_open = 0;
                }
            } else {
                pos_a = a[ia * 2 + 1];
                ia++;
                while (ia < a_count && pos_a >= a[ia * 2 + 0])
                    pos_a = a[ia++ * 2 + 1];
                a_open = 0;
            }
        } else {
            if (b_open == 0) {
                if (ib < b_count) {
                    pos_b = b[ib * 2 + 0];
                    b_open = 1;
                } else {
                    pos_b = 0x281;
                    b_open = 0;
                }
            } else {
                pos_b = b[ib * 2 + 1];
                ib++;
                while (ib < b_count && pos_b >= b[ib * 2 + 0])
                    pos_b = b[ib++ * 2 + 1];
                b_open = 0;
            }
        }
    }

    if (*(uint16_t *)(out + 0x28) == 2)
        *(uint16_t *)(out + 0x28) = 0;
}
